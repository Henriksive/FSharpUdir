							module GetProveQuery = 
								open FSharp.Data
								open UDIR.Prover.Models    
								type private Query = SqlCommandProvider<"
									SELECT
										[Lopenummer],
										[Fag],
										[Trinn]
									FROM
										[Prove].[Prove]
									WHERE
										Lopenummer = @blabla"
									, DbName
									, SingleRow = true>
						

								let RunAsync(proveId: int) = 
									async { 
										let! result = (Query.Create()).AsyncExecute(proveId)
										return
											result
											|> Option.map (fun e -> 
												new Prove(
													e.Id,
													e.Fag,
													e.Trinn))
											|> OptionUtils.getOrNull
									}
									|> Async.StartAsTask
